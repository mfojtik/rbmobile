    def submit(label=nil, opts={})
      opts = {
        'data-theme''data-theme' => opts.delete(:theme)
      }.merge(opts)
      capture_haml do
        form_field do
          haml_tag :button, { :type => :submit }.merge(opts), :value => label do
            haml_concat label || 'Submit'
          end
        end
      end
    end
