    def item(opts={}, &block)
      opts['data-icon''data-icon'] = opts.delete(:icon) if opts[:icon]
      opts['data-theme''data-theme'] = opts.delete(:theme) if opts[:theme]
      haml_tag :li, opts do
        block.call if block_given?
        haml_tag(:img, :src => opts[:item_icon_url], :class => 'ui-li-icon') if opts[:item_icon_url]
      end
    end
