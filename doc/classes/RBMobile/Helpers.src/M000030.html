    def form(url, method=:post, opts={}, &block)
      opts.merge!(
        :method => method,
        :url => url
      )
      opts.merge!('data-ajax''data-ajax' => 'false') if not RBMobile::config[:ajax] and not opts.delete(:ajax)
      haml_tag :form, opts do
        block.call if block_given?
      end
    end
