    def button(kind, url, label, opts={})
      options = {
        'data-icon''data-icon' => kind,
        'data-role''data-role' => :button,
        :href => url
      }.merge(opts)
      options.merge!('data-ajax''data-ajax' => 'false') if not RBMobile::config[:ajax] and not options.delete(:ajax)
      options.merge!('data-theme''data-theme' => options.delete(:theme)) if options[:theme]
      capture_haml do
        haml_tag :a, options do
          haml_concat label
        end
      end
    end
