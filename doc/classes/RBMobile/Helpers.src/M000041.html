    def select(name, label=nil, options=[], opts={})
      opts = {
        'data-native-menu''data-native-menu' => opts.delete(:native) ? 'true' : 'false',
        'data-theme''data-theme' => opts.delete(:theme)
      }.merge(opts)
      capture_haml do
        form_field do
          haml_tag :label, :for => name do
            haml_concat label || name.to_s.capitalize
          end
          haml_tag :select, { :id => name }.merge(opts) do
            options.each do |option|
              option = [option] unless option.kind_of? Array
              # One-item array will create a placeholder
              haml_tag :option, :value => option.size==2 ? option.first : nil, 'data-placeholder''data-placeholder' => option.size==1 ? 'true' : nil do
                haml_concat option.last
              end
            end
          end
        end
      end
    end
