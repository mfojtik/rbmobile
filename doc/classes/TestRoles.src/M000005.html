  def test_it_generates_basic_header_role
    output = capture_haml do
      page do
        header do
        end
      end
    end
    retval = Nokogiri::HTML(output)
    assert_equal retval.css("div[@data-role=page] div").size, 1
    assert_equal retval.css("div[@data-role=page] div").first['data-role''data-role'], "header"
    assert_equal retval.css("div[@data-role=page] div").first['data-position''data-position'], "inline"
  end
